1.
sudo nano /etc/apache2/sites-available/000_default.conf (remove /public on DocumentRoot)


2.
sudo service apahce2 restart


3.
mkdir ~/.npm-global && npm config set prefix '~/.npm-global' && export PATH=~/.npm-global/bin:$PATH && source ~/.profile


4.
mkdir ~/.npm-global/lib

4.1.
mkdir ~/.npm-global/lib/node_modules

4.2.
mkdir ~/.npm-global/bin

5.

sudo npm install cross-env webpack webpack-dev-server --save


8.
npm run dev
